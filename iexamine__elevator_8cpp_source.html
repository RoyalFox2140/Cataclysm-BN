93auto vehicle_status( const wrapped_vehicle &veh, const elevator::tiles &tiles ) -> overlap_status
94{
95 const auto &ps = veh.v->get_all_parts();
96 const int all_vparts_count = ps.part_count();
97 const int vparts_inside = std::count_if( ps.begin(), ps.end(), [&]( const vpart_reference & vp ) {
98 const tripoint p = veh.pos + vp.part().precalc[0];
99 const auto eit = std::find( tiles.cbegin(), tiles.cend(), p );
100 return eit != tiles.cend();
101 } );
102
103 if( vparts_inside == all_vparts_count ) {
104 return overlap_status::inside;
105 } else if( vparts_inside == 0 ) {
106 return overlap_status::outside;
107 } else {
108 return overlap_status::overlap;
109 }
